ChainedScatterObject ChainedScatterTable::find(char* key, int value)
{
	unsigned int probe = h(key);
	while (probe != -1)
	{
		if (!array[probe].isEqualKey(key) && value != array[probe].value)
			probe = array[probe].next;
		else
			return array[probe];
	}
	return ChainedScatterObject();
}
